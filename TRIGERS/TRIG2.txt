CREATE OR REPLACE TRIGGER TRIG1 BEFORE DELETE OR INSERT OR UPDATE ON EMP_TAB FOR EACH ROW
BEGIN
IF DELETING THEN INSERT INTO EMP_X VALUES(:OLD.ENO,:OLD.ENAME,:OLD.ESAL,:OLD.JOB,SYSDATE);
ELSIF 
INSERTING THEN INSERT INTO EMP_NEW VALUES(:NEW.ENO,:NEW.ENAME,:NEW.ESAL,:NEW.JOB);
ELSE
INSERT INTO EMP_MODIFIED VALUES(:OLD.ENO,:OLD.ENAME,:OLD.ESAL,:OLD.JOB);
END IF;
END;
